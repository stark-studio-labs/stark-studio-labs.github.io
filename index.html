my_mod/events.py 
from stark.events import EventBus, Event
from stark.inject import override
from stark.settings import ModSettings
from dataclasses import dataclass


@dataclass
class SimPromotedEvent(Event):
    """Fired when a Sim earns a career promotion."""
    sim_id: int
    career: str
    new_level: int
    salary_increase: float


bus = EventBus()
settings = ModSettings("stark_economy")


@bus.on(SimPromotedEvent)
def handle_promotion(event: SimPromotedEvent):
    # Adjust neighborhood economy based on promotion
    multiplier = settings.get("economy.promotion_multiplier", 1.2)
    adjust_local_economy(event.sim_id, event.salary_increase * multiplier)
    bus.emit(EconomyShiftEvent(
        neighborhood=get_neighborhood(event.sim_id),
        delta=event.salary_increase * multiplier
    ))


@override("career.promote")
def on_career_promote(sim, career, original_fn):
    result = original_fn(sim, career)
    bus.emit(SimPromotedEvent(
        sim_id=sim.id,
        career=career.name,
        new_level=career.level,
        salary_increase=career.salary - career.prev_salary
    ))
    return result
