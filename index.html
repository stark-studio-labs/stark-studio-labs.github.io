from stark_framework import mod, on_event
from stark_framework.sims import Sim

@mod("confidence_boost", version="1.0.0")
class ConfidenceMod:

    @on_event("sim.interaction.start")
    def on_interact(self, sim: Sim):
        if sim.interaction.is_social:
            sim.add_buff("confident", duration=240)

    @on_event("sim.skill.level_up")
    def on_skill(self, sim: Sim, skill, level: int):
        self.log(f"{sim.name} reached {skill} {level}")
